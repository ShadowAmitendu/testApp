Pipes Example

Original:

{{ firstName }} {{ middleName }} {{ lastName }}

Uppercase:

{{ firstName | uppercase }} {{ middleName | uppercase }} {{ lastName | uppercase }}

Lowercase:

{{ firstName | lowercase }} {{ middleName | lowercase }} {{ lastName | lowercase }}

Titlecase:

{{ firstName | titlecase }} {{ middleName | titlecase }} {{ lastName | titlecase }}

Date Pipe:

Original: {{ date2 }}

Default: {{ date2 | date }}

Custom: {{ date2 | date : "dd-MM-yyyy" }}

Currency Pipe:

{{ 12345.678 | currency : "INR" : "symbol" : "1.2-2" }}

Percent Pipe:

{{ 0.857 | percent : "1.1-2" }}

Number Pipe:

{{ 12345.6789 | number : "1.2-3" }}

Slice Pipe:

{{ firstName | slice : 0 : 5 }}

JSON Pipe:
{{ { name: firstName, age: 25, city: "Kolkata" } | json }}
Async Pipe (Observable):

{{ asyncData$ | async }}

Async Pipe (Promise):

{{ asyncPromise | async }}
